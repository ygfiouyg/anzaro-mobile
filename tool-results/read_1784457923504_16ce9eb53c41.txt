   550→      sendMessage: async (content: string, attachments?: File[], forceSearch?: boolean) => {
   551→        const {
   552→          activeConversationId,
   553→          activeModel,
   554→          activeLanguage,
   555→          autoWebSearch,
   556→          systemPromptMode,
   557→          createConversation,
   558→          addMessage,
   559→        } = get();
   560→
   561→        // ─── Safety guard: Reset stuck isStreaming AND abort previous stream ────
   562→        // If isStreaming was left true from a previous failed attempt, force-reset it
   563→        // Also abort any in-flight stream to prevent resource leaks and race conditions
   564→        if (get().isStreaming) {
   565→          console.warn('[Chat] isStreaming was stuck — force resetting before new message');
   566→          // Abort the previous stream's fetch so its reader stops consuming
   567→          if (activeStreamAbortController) {
   568→            try {
   569→              activeStreamAbortController.abort();
   570→              console.warn('[Chat] Aborted previous stream AbortController');
   571→            } catch { /* ignore */ }
   572→            activeStreamAbortController = null;
   573→          }
   574→          set({ isStreaming: false, streamingProgress: null });
   575→        }
   576→
   577→        // Get or create conversation
   578→        let conversationId = activeConversationId;
   579→        if (!conversationId) {
   580→          const conv = createConversation();
   581→          conversationId = conv.id;
   582→        }
   583→
   584→        const now = new Date().toISOString();
   585→
   586→        // Strip large base64 data from content for local storage to prevent
   587→        // localStorage overflow. Keep the full content for the API request.
