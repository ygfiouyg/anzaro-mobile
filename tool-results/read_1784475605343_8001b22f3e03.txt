    50→/**
    51→ * Build the messages array for the LLM, handling multimodal content for images.
    52→ */
    53→async function buildLLMMessages(
    54→  systemPrompt: string,
    55→  conversationMessages: { role: string; content: string }[],
    56→  userMessage: string,
    57→  parsed: { cleanedMessage: string; attachments: ParsedAttachment[]; hasAttachments: boolean },
    58→  glmModel: string,
    59→  modelConfig: { provider: string; capabilities: { vision: boolean } }
    60→): Promise<Array<{ role: string; content: string | Array<{ type: string; text?: string; image_url?: { url: string } }> }>> {
    61→  const isVisionModel = glmModel === 'glm-4v-flash' || glmModel === 'glm-4v' || glmModel === 'glm-4-flash' || (modelConfig.provider === 'gemini' && modelConfig.capabilities.vision) || (modelConfig.provider === 'zhipuai' && modelConfig.capabilities.vision);
    62→  const imageAttachments = parsed.attachments.filter((a) => a.type === 'image');
    63→  const pdfAttachments = parsed.attachments.filter((a) => a.type === 'pdf');
    64→  const docxAttachments = parsed.attachments.filter((a) => a.type === 'docx');
    65→
    66→  // Build the user message content
    67→  let finalUserMessage = parsed.cleanedMessage;
    68→
    69→  // For PDFs, extract text and append it
    70→  if (pdfAttachments.length > 0) {
    71→    const pdfTexts = await Promise.all(
    72→      pdfAttachments.map(async (pdf) => {
    73→        const extractedText = await extractPdfWithVlmAndText(pdf.content!, pdf.name);
    74→        const contentQuality = classifyContentQuality(extractedText);
    75→        if (contentQuality === 'failed') {
    76→          return `📄 ملف PDF مرفق: ${pdf.name} (${pdf.size})\n⚠️ ${extractedText}\n⚠️ مهم: لا تخترع أي محتوى عن هذا الملف — لم يتم قراءته بنجاح. أخبر المستخدم بذلك بصراحة.`;
    77→        }
    78→        if (contentQuality === 'partial') {
    79→          // Extract usable content after the failure marker
    80→          const firstBracketEnd = extractedText.indexOf(']\n');
    81→          const usableText = firstBracketEnd > 0 ? extractedText.slice(firstBracketEnd + 2).trim() : extractedText;
    82→          return `📄 ملف PDF مرفق: ${pdf.name} (${pdf.size})\n⚠️ محتوى جزئي — بعض الصفحات لم يتم قراءتها بشكل صحيح\n--- محتوى PDF ---\n${usableText}\n--- نهاية المحتوى ---`;
    83→        }
    84→        return `📄 ملف PDF مرفق: ${pdf.name} (${pdf.size})\n--- محتوى PDF ---\n${extractedText}\n--- نهاية المحتوى ---`;
    85→      })
    86→    );
    87→    finalUserMessage = pdfTexts.join('\n\n') + (finalUserMessage ? '\n\n' + finalUserMessage : '');
    88→  }
    89→
    90→  // For DOCX (Word) files, extract text via mammoth and append it
    91→  if (docxAttachments.length > 0) {
    92→    const docxTexts = await Promise.all(
    93→      docxAttachments.map(async (docx) => {
    94→        try {
    95→          const extractedText = await extractTextFromDocxBase64(docx.content!, 100 * 1024);
    96→          if (!extractedText || extractedText.length < 10) {
    97→            return `📄 ملف Word مرفق: ${docx.name} (${docx.size})\n⚠️ لم يتم استخراج نص من الملف. أخبر المستخدم بذلك بصراحة.`;
    98→          }
    99→          return `📄 ملف Word مرفق: ${docx.name} (${docx.size})\n--- محتوى الملف ---\n${extractedText}\n--- نهاية المحتوى ---`;
   100→        } catch (err) {
   101→          return `📄 ملف Word مرفق: ${docx.name} (${docx.size})\n⚠️ خطأ في قراءة الملف: ${err instanceof Error ? err.message : String(err)}`;
   102→        }
   103→      })
   104→    );
   105→    finalUserMessage = docxTexts.join('\n\n') + (finalUserMessage ? '\n\n' + finalUserMessage : '');
   106→  }
   107→
   108→  // If we have images and a vision model, construct multimodal messages
   109→  if (imageAttachments.length > 0 && isVisionModel) {
   110→    // Build multimodal content array
   111→    const contentParts: Array<{ type: string; text?: string; image_url?: { url: string } }> = [];
   112→
   113→    // Add text content first
   114→    if (finalUserMessage) {
   115→      contentParts.push({ type: 'text', text: finalUserMessage });
   116→    }
   117→
   118→    // Add each image
   119→    for (const img of imageAttachments) {
   120→      contentParts.push({
   121→        type: 'image_url',
   122→        image_url: { url: img.content! },
   123→      });
   124→    }
   125→
   126→    return [
   127→      { role: 'system', content: systemPrompt },
   128→      ...conversationMessages,
   129→      { role: 'user', content: contentParts },
   130→    ];
   131→  }
   132→
   133→  // For non-vision models with images, add a text description note
   134→  if (imageAttachments.length > 0 && !isVisionModel) {
   135→    // ── Media Preprocessor: Enhanced media analysis ──
   136→    // If media preprocessing is available, use it for richer content analysis
   137→    // This enables ALL models to understand images, videos, and audio
   138→    // through preprocessing with vision/ASR services.
   139→    try {
   140→      const mediaAttachments: ParsedMediaAttachment[] = imageAttachments.map((img) => ({
   141→        type: 'image' as const,
   142→        name: img.name,
   143→        size: img.size,
   144→        content: img.content,
   145→      }));
   146→      const preprocessed = await preprocessMediaAttachments(
   147→        mediaAttachments,
   148→        finalUserMessage,
   149→        false, // not a vision model
   150→        'ar'
   151→      );
   152→      if (preprocessed.combinedText.trim()) {
   153→        finalUserMessage = preprocessed.combinedText + (finalUserMessage ? '\n\n' + finalUserMessage : '');
   154→      }
   155→    } catch (mediaPreprocessErr) {
   156→      // Fallback: tell the user to switch to a vision model
   157→      console.warn('[Chat] Media preprocessor failed, using fallback note:', mediaPreprocessErr instanceof Error ? mediaPreprocessErr.message : String(mediaPreprocessErr));
   158→      const imageNote = imageAttachments
   159→        .map((img) => `📷 صورة مرفقة: ${img.name} (${img.size}) - تم إرفاق صورة لكن النموذج الحالي لا يدعم تحليل الصور. يرجى التبديل لنموذج Delta Vision لتحليل الصور.`)
   160→        .join('\n');
   161→      finalUserMessage = imageNote + (finalUserMessage ? '\n\n' + finalUserMessage : '');
   162→    }
   163→  }
   164→
   165→  // Standard text-only messages
   166→  return [
   167→    { role: 'system', content: systemPrompt },
   168→    ...conversationMessages,
   169→    { role: 'user', content: finalUserMessage },
   170→  ];
   171→}

   175→// ─── POST Handler ────────────────────────────────────────────────────
   176→export async function POST(request: NextRequest) {
   177→  // ── Rate limiting for chat stream (most resource-intensive endpoint) ──
   178→  const { checkRateLimit, RATE_LIMIT_PRESETS } = await import('@/lib/rate-limit');
   179→  const rateLimitResponse = checkRateLimit(request, RATE_LIMIT_PRESETS.ai);
   180→  if (rateLimitResponse) return rateLimitResponse;
   181→
   182→  let connectionId = '';
   183→  try {
   184→    const body = await request.json();
   185→    const { message, model, language, conversationId, autoSearch, forceSearch, systemPromptMode } = body as {
   186→      message: string;
   187→      model: string;
   188→      language: string;
   189→      conversationId?: string;
   190→      autoSearch?: boolean;
   191→      forceSearch?: boolean;
   192→      systemPromptMode?: 'full' | 'open';
   193→    };
   194→
   195→    // Validate required fields
   196→    if (!message || !model) {
   197→      return new Response(
   198→        JSON.stringify({ error: 'الرسالة والنموذج مطلوبان' }),
   199→        { status: 400, headers: { 'Content-Type': 'application/json' } }
   200→      );
   201→    }
   202→
   203→    // ── MCP Tools Integration ──
   204→    // اكتشف نية المستخدم وشغّل أداة MCP لو محتاجة
   205→    // قبل ما نبعت لـ GLM
   206→    // CRITICAL: Skip MCP detection if message contains image/file attachments
   207→    // because base64 data can contain substrings like "acp" that falsely match
   208→    // MCP tool patterns (e.g., agent-acp), intercepting the message and preventing
   209→    // the image from reaching the vision pipeline.
   210→    const hasEmbeddedAttachments = message.includes('[DELTA_IMAGE:') || message.includes('[DELTA_PDF:') || message.includes('[DELTA_DOCX:');
   211→    
   212→    if (!hasEmbeddedAttachments) {
   213→      try {
   214→        const { detectAndRunMCP, detectAndRunVision } = await import('@/lib/ai-tools/mcp-chat-integration');
   215→        const mcpResult = await detectAndRunMCP(message);
   216→
   217→        if (mcpResult.matched && mcpResult.result) {
   218→          console.log(`[Chat] MCP tool matched: ${mcpResult.tool}`);
   219→          const mcpStream = new ReadableStream({
   220→            async start(controller) {
   221→              const encoder = new TextEncoder();
   222→              controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: mcpResult.result })}\n\n`));
   223→              controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   224→              controller.close();
   225→            },
   226→          });
   227→          return new Response(mcpStream, {
   228→            headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache, no-transform', Connection: 'keep-alive', 'X-Accel-Buffering': 'no' },
   229→          });
   230→        }
   231→      } catch (mcpError) {
   232→        console.warn('[Chat] MCP detection failed:', mcpError);
   233→      }
   234→    }
   235→
   236→    // ── Smart Ball Command Detection (Reversed Command Control) ──
   237→    // لو المستخدم طلب تشغيل/إيقاف راديو، تحكم في جهاز، أو تفعيل مشهد مزاجي
   238→    // نفّذ الأمر فوراً عبر control-engine وأرجع تأكيد
   239→    if (!hasEmbeddedAttachments) {
   240→      try {
   241→        const { detectSmartBallCommand } = await import('@/lib/anzaro-smart-ball-detector');
   242→        const ballCommand = await detectSmartBallCommand(message);
   243→        if (ballCommand) {
   244→          console.log(`[Chat] Smart Ball command detected: ${ballCommand.type}`);
   245→          const ballStream = new ReadableStream({
   246→            async start(controller) {
   247→              const encoder = new TextEncoder();
   248→              // sink accepts either a string (text content) or an object (structured SSE event)
   249→              const sink = (data: string | Record<string, unknown>) => {
   250→                if (typeof data === 'string') {
   251→                  controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: data })}\n\n`));
   252→                } else {
   253→                  controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`));
   254→                }
   255→              };
   256→              try {
   257→                await ballCommand.execute(sink);
   258→              } catch (e: any) {
   259→                sink(`\n\n❌ خطأ: ${e.message}`);
   260→              } finally {
   261→                controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   262→                controller.close();
   263→              }
   264→            },
   265→          });
   266→          return new Response(ballStream, {
   267→            headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache, no-transform', Connection: 'keep-alive', 'X-Accel-Buffering': 'no' },
   268→          });
   269→        }
   270→      } catch (ballError) {
   271→        console.warn('[Chat] Smart Ball detection failed:', ballError);
   272→      }
   273→    }
   274→
   275→    // ── Intent Detection (Script Writer + Content Studio) ──
   276→    // لو المستخدم طلب سكريبت أو حزمة محتوى، حوّل للأداة المناسبة
   277→    try {
   278→      const { detectIntent } = await import('@/lib/intent/router');
   279→      const intent = detectIntent(message);
   280→      if (intent.matched && intent.confidence === 'high') {
   281→        console.log(`[Chat] Intent matched: ${intent.tool} — ${intent.contentType} — topic: ${intent.topic}`);
   282→
   283→        const isContentStudio = intent.tool === 'content-studio';
   284→        const moduleName = isContentStudio ? '@/lib/content-studio/engine' : '@/lib/scriptwriter/engine';
   285→        const funcName = isContentStudio ? 'generateContentPackage' : 'generateScript';
   286→        const importedModule: any = await import(moduleName);
   287→        const generator = importedModule[funcName];
   288→
   289→        const toolStream = new ReadableStream({
   290→          async start(controller) {
   291→            const encoder = new TextEncoder();
   292→            const sink = (event: any) => {
   293→              if (event.type === 'token') {
   294→                controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: event.content })}\n\n`));
   295→              } else if (event.type === 'done') {
   296→                controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   297→              } else if (event.type === 'error') {
   298→                controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: '\n\n❌ خطأ: ' + event.error })}\n\n`));
   299→                controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   300→              }
   301→              // status/thinking/studio_done/script_done events silently dropped (HF stream format)
   302→            };
   303→            try {
   304→              await generator({
   305→                topic: intent.topic || message,
   306→                contentType: intent.contentType || 'reel',
   307→                language: 'ar',
   308→              }, sink);
   309→            } catch (e: any) {
   310→              controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: '\n\n❌ خطأ: ' + e.message })}\n\n`));
   311→              controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   312→            } finally {
   313→              controller.close();
   314→            }
   315→          },
   316→        });
   317→
   318→        return new Response(toolStream, {
   319→          headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache, no-transform', Connection: 'keep-alive', 'X-Accel-Buffering': 'no' },
   320→        });
   321→      }
   322→    } catch (intentError) {
   323→      console.warn('[Chat] Intent detection failed:', intentError);
   324→    }
   325→
   326→    // ── Media Intent Detection (Radio / Spotify / YouTube / TTS) ──
   327→    // CONTEXT-BASED detection: uses a fast LLM call to understand the user's
   328→    // INTENT from natural Arabic phrasing — not keyword-regex matching.
   329→    //
   330→    // This handles phrases like:
   331→    //   "شغللي آخر حاجة القناة دي نزلتها"  → youtube (no keyword "فيديو")
   332→    //   "سمعني اللي كان طالع"              → youtube (context: was playing)
   333→    //   "خليني أسمع حاجة هادية"             → radio (context: calm)
   334→    //
   335→    // Falls back to regex ONLY if the LLM call fails (network/timeout).
   336→    try {
   337→      const { detectMediaIntentLLM, detectMediaIntentRegex } = await import('@/lib/ai-tools/media-intent-llm');
   338→
   339→      // Try LLM-based detection first (context-aware)
   340→      let intent = await detectMediaIntentLLM(message);
   341→
   342→      // Fallback to regex if LLM failed
   343→      if (!intent.wantsMedia && !intent.source) {
   344→        const fallback = detectMediaIntentRegex(message);
   345→        if (fallback.wantsMedia) {
   346→          console.log('[Chat] LLM detection failed, regex fallback triggered');
   347→          intent = fallback;
   348→        }
   349→      }
   350→
   351→      // ── V.15: STOP media intent — send stopMedia SSE event ──
   352→      if (intent.wantsMedia && intent.action === 'stop') {
   353→        console.log('[Chat] Media STOP intent detected');
   354→        const stopStream = new ReadableStream({
   355→          async start(controller) {
   356→            const encoder = new TextEncoder();
   357→            controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: 'تمام، اتقفل 🔇' })}\n\n`));
   358→            controller.enqueue(encoder.encode(`data: ${JSON.stringify({ stopMedia: true })}\n\n`));
   359→            controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   360→            controller.close();
   361→          },
   362→        });
   363→        return new Response(stopStream, {
   364→          headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache, no-transform', Connection: 'keep-alive', 'X-Accel-Buffering': 'no' },
   365→        });
   366→      }
   367→
   368→      if (intent.wantsMedia && intent.source && intent.source !== 'tts') {
   369→        console.log(`[Chat] Media intent (LLM): source=${intent.source}, query="${(intent.query || message).slice(0, 60)}", confidence=${intent.confidence || 0}`);
   370→
   371→        // Call play-media API internally
   372→        const mediaResponse = await fetch('http://localhost:3000/api/ai/play-media', {
   373→          method: 'POST',
   374→          headers: { 'Content-Type': 'application/json' },
   375→          body: JSON.stringify({ query: intent.query || message, source: intent.source }),
   376→          signal: AbortSignal.timeout(15_000),
   377→        });
   378→
   379→        if (mediaResponse.ok) {
   380→          const mediaData = await mediaResponse.json();
   381→          if (mediaData.mediaWidget) {
   382→            console.log(`[Chat] Media widget ready: source=${mediaData.mediaWidget.source}, title="${mediaData.mediaWidget.title}"`);
   383→
   384→            // Return SSE stream with text + mediaWidget
   385→            const mediaStream = new ReadableStream({
   386→              async start(controller) {
   387→                const encoder = new TextEncoder();
   388→                controller.enqueue(encoder.encode(`data: ${JSON.stringify({ content: mediaData.content })}\n\n`));
   389→                controller.enqueue(encoder.encode(`data: ${JSON.stringify({ mediaWidget: mediaData.mediaWidget })}\n\n`));
   390→                controller.enqueue(encoder.encode('data: [DONE]\n\n'));
   391→                controller.close();
   392→              },
   393→            });
   394→
   395→            return new Response(mediaStream, {
   396→              headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache, no-transform', Connection: 'keep-alive', 'X-Accel-Buffering': 'no' },
   397→            });
   398→          }
   399→        }
   400→      }
   401→    } catch (mediaError) {
   402→      console.warn('[Chat] Media detection failed:', mediaError);
   403→    }
   404→
   405→    // ── HuggingFace Chat Model Bridge ──
   406→    let modelConfig = getModelById(model);
   407→    let isHFDirectModel = false;
   408→    let hfDirectModelId: string | null = null;
   409→
   410→    if (!modelConfig && model?.startsWith('hf-chat:')) {
   411→      hfDirectModelId = model.slice(8); // Strip 'hf-chat:' prefix
   412→
   413→      // Check if model is disabled by admin
   414→      if (await isModelDisabled(hfDirectModelId)) {
   415→        return new Response(
   416→          JSON.stringify({ error: `النموذج "${hfDirectModelId}" معطل من قبل الآدمن` }),
   417→          { status: 403, headers: { 'Content-Type': 'application/json' } }
   418→        );
   419→      }
   420→
   421→      const hfEntry = getChatModelById(hfDirectModelId);
   422→      if (hfEntry) {
   423→        // Create a synthetic modelConfig for the HF model
   424→        modelConfig = {
   425→          id: model,
   426→          name: hfEntry.name || hfDirectModelId,
